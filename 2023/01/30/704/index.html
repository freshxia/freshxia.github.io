题目：

用二分查找搜索元素

代码：

class Solution {
public:
    int search(vector<int>& nums, int target) {
        int n = nums.size();
        int mid,l=0,r=n-1;
        while(l<=r){
            mid=(r+l)/2;
        if(nums[mid]<target) {
          l=mid+1;
        }
        else if(nums[mid]>target) {
            r=mid-1;
        } 
        else return mid;
        }
        return -1;
    }
};
